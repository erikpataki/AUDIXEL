Examples
// Confirmation dialog
<Modal
  isOpen={showConfirmDialog}
  onClose={handleCancel}
  onConfirm={handleConfirm}
  title="Confirm Action"
  message="Are you sure you want to continue?"
/>
// Tutorial modal
<Modal
  isOpen={showTutorial}
  onClose={closeTutorial}
  title="How to Use"
  message={tutorialContent}
  modalType="tutorial"
  hasButtons={false}
  customClass="tutorial-modal"
/>
